39  InputIt end,
40  const std::string & separator,
41  const std::string & concluder)
42 {
43  std::ostringstream ss;
44 
45  if(begin != end) {

146

150  std::ostringstream oss;

306 
308  WireFormatElement& operator=(const std::string &value){
309  static std::map<std::string, WireFormat> lookup = {
310  {"stomp", WireFormat::stomp},
311  {"openwire", WireFormat::openwire}
312  };
313 
314  try {
315  std::string wireFormat = value;
316  boost::algorithm::to_lower(wireFormat);
317  InternalElement<WireFormat>::operator=(lookup.at(wireFormat));
318  } catch(std::out_of_range& e) {
319  std::string msg("Unknown wire format string (");
320  msg += value;
321  msg += ")";
322  throw offcenter::common::RuntimeException(msg);
323  }
324 
325  return *this;
326  }

330  return operator=(value);
331  }
